# PT 알림톡 발송 시스템 구현 계획

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** boot.soritune.com의 알림톡 발송 시스템을 pt.soritune.com에 이식한다. 시트 기반 캠페인을 PT members.phone으로 발송하는 합성 어댑터를 핵심으로 한다.

**Architecture:** boot 코드를 변경 없이 복사(`solapi_client`/`notify_functions`/`scenario_registry`/`source_google_sheet`) + 1줄만 수정(`dispatcher` source dispatch에 PT 어댑터 케이스 추가) + PT 신규(`source_pt_sheet_member` 어댑터, PT auth로 변환된 API, SPA admin 페이지). 4개 boot 테이블 + PT 신규 1개 테이블.

**Tech Stack:** PHP 8 (no framework), MySQL/PDO, Google Sheets API (Service Account JSON), Solapi REST API, vanilla JS SPA pattern (PT 컨벤션).

**Spec:** `docs/superpowers/specs/2026-04-29-pt-notify-design.md`

**작업 디렉토리:** `/var/www/html/_______site_SORITUNECOM_PT/` (PT 운영 사이트, DEV/PROD 분리 없음). PT는 main 단일 브랜치.

## Task 1: 4개 boot 테이블 마이그레이션

**Files:**

- [ ] **Step 1: 마이그 SQL 작성**

`migrations/20260429_add_notify_tables.sql` 파일 생성, 내용:

```sql
-- PT 알림톡: boot의 4개 핵심 테이블 이식
-- 출처: boot-dev/migrate_notify_tables.php의 SQL을 그대로 복사
-- 적용: mysql -u <user> -p <db> < migrations/20260429_add_notify_tables.sql

CREATE TABLE IF NOT EXISTS notify_scenario_state (
    scenario_key      VARCHAR(64)  NOT NULL PRIMARY KEY,
    is_active         TINYINT(1)   NOT NULL DEFAULT 0,
    is_running        TINYINT(1)   NOT NULL DEFAULT 0,
    running_since     DATETIME     NULL,
    last_run_at       DATETIME     NULL,
    last_run_status   VARCHAR(20)  NULL,
    last_batch_id     BIGINT       NULL,
    notes             TEXT         NULL,
    updated_at        TIMESTAMP    DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    updated_by        VARCHAR(64)  NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE IF NOT EXISTS notify_batch (
    id              BIGINT AUTO_INCREMENT PRIMARY KEY,
    scenario_key    VARCHAR(64)  NOT NULL,
    trigger_type    ENUM('schedule','manual','retry') NOT NULL,
    triggered_by    VARCHAR(64)  NULL,
    started_at      DATETIME     NOT NULL,
    finished_at     DATETIME     NULL,
    target_count    INT          NOT NULL DEFAULT 0,
    sent_count      INT          NOT NULL DEFAULT 0,
    failed_count    INT          NOT NULL DEFAULT 0,
    unknown_count   INT          NOT NULL DEFAULT 0,
    skipped_count   INT          NOT NULL DEFAULT 0,
    dry_run         TINYINT(1)   NOT NULL DEFAULT 0,
    status          ENUM('running','completed','partial','failed','no_targets') NOT NULL,
    error_message   TEXT         NULL,
    INDEX idx_scenario_started (scenario_key, started_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE IF NOT EXISTS notify_message (
    id                BIGINT AUTO_INCREMENT PRIMARY KEY,
    batch_id          BIGINT       NOT NULL,
    scenario_key      VARCHAR(64)  NOT NULL,
    row_key           VARCHAR(255) NOT NULL,
    phone             VARCHAR(20)  NOT NULL,
    name              VARCHAR(64)  NULL,
    template_id       VARCHAR(64)  NOT NULL,
    rendered_text     TEXT         NULL,
    channel_used      ENUM('alimtalk','lms','none') NOT NULL DEFAULT 'none',
    status            ENUM('queued','sent','failed','skipped','dry_run','unknown') NOT NULL DEFAULT 'queued',
    skip_reason       VARCHAR(64)  NULL,
    fail_reason       TEXT         NULL,
    solapi_message_id VARCHAR(64)  NULL,
    sent_at           DATETIME     NULL,
    processed_at      DATETIME     NULL,
    created_at        TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_cooldown (scenario_key, phone, status, processed_at),
    INDEX idx_batch (batch_id),
    INDEX idx_solapi (solapi_message_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE IF NOT EXISTS notify_preview (
    id            CHAR(32)     NOT NULL PRIMARY KEY,
    scenario_key  VARCHAR(64)  NOT NULL,
    dry_run       TINYINT(1)   NOT NULL,
    row_keys      JSON         NOT NULL,
    target_count  INT          NOT NULL,
    created_by    VARCHAR(64)  NOT NULL,
    created_at    DATETIME     NOT NULL,
    expires_at    DATETIME     NOT NULL,
    used_at       DATETIME     NULL,
    INDEX idx_expires (expires_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
```

 *   [['row_key', 'phone', 'name', 'columns', 'match_status', 'resolved_member_id', 'sheet_row_idx', 'soritune_id'], ...]

 *   - 'phone_empty'       — DB에는 있으나 phone NULL/빈값

const PT_NOTIFY_MERGE_MAX_HOPS = 5;

function notifySourcePtSheetMember(array $cfg): array {
    $required = ['sheet_id', 'tab', 'range', 'check_col', 'soritune_id_col', 'name_col'];
    foreach ($required as $r) {
        if (empty($cfg[$r])) {
            throw new RuntimeException("source.{$r} 누락");
        }
    }
    $checkValue = isset($cfg['check_value']) && $cfg['check_value'] !== ''
        ? (string)$cfg['check_value']
        : 'Y';

    $sheet = new GoogleSheets();
    $rangeFull = $cfg['tab'] . '!' . $cfg['range'];
    $values = $sheet->getValues($cfg['sheet_id'], $rangeFull);
    if (!is_array($values) || count($values) < 2) return [];

    return notifyPtResolveSheetRows($values, $cfg, $checkValue);
}

/**
 * 시트 데이터(2D array) + cfg를 받아 lookup 결과 행 배열을 반환.
 * Google Sheets 호출과 분리하여 단위 테스트 가능하게 함.
 */
function notifyPtResolveSheetRows(array $values, array $cfg, string $checkValue): array {
    $headers = array_map('strval', $values[0]);
    $headerIdx = array_flip($headers);

    foreach ([$cfg['check_col'], $cfg['soritune_id_col'], $cfg['name_col']] as $h) {
        if (!isset($headerIdx[$h])) {
            throw new RuntimeException("시트에 '{$h}' 헤더가 없습니다");
        }
    }

    $checkIdx = $headerIdx[$cfg['check_col']];
    $sidIdx = $headerIdx[$cfg['soritune_id_col']];

    $db = getDB();
    $stmt = $db->prepare("SELECT id, phone, merged_into FROM members WHERE soritune_id = ? LIMIT 1");

    $results = [];
    $rowCount = count($values);
    for ($i = 1; $i < $rowCount; $i++) {
        $row = $values[$i];
        $checkVal = isset($row[$checkIdx]) ? trim((string)$row[$checkIdx]) : '';
        if (strcasecmp($checkVal, $checkValue) !== 0) continue;

        $columns = [];
        foreach ($headers as $j => $h) {
            $columns[$h] = isset($row[$j]) ? (string)$row[$j] : '';
        }
        $sid = trim((string)($columns[$cfg['soritune_id_col']] ?? ''));

        $resolved = notifyPtResolveMember($stmt, $sid);

        $results[] = [
            'row_key'             => sprintf('pt:%s:%s:%d', $cfg['sheet_id'], $cfg['tab'], $i + 1),
            'phone'               => $resolved['phone'] ?? '',
            'name'                => $columns[$cfg['name_col']] ?? '',
            'columns'             => $columns,
            'match_status'        => $resolved['match_status'],
            'resolved_member_id'  => $resolved['member_id'],
            'sheet_row_idx'       => $i + 1,
            'soritune_id'         => $sid,
        ];
    }
    return $results;
}

/**
 * soritune_id로 members lookup. merged_into가 있으면 마스터로 추적 (최대 PT_NOTIFY_MERGE_MAX_HOPS 회).
 * 반환: ['member_id'=>?, 'phone'=>?, 'match_status'=>...]
 */
function notifyPtResolveMember(PDOStatement $stmt, string $soritune_id): array {
    if ($soritune_id === '') {
        return ['member_id' => null, 'phone' => null, 'match_status' => 'member_not_found'];
    }
    $stmt->execute([$soritune_id]);
    $row = $stmt->fetch(PDO::FETCH_ASSOC);
    if (!$row) {
        return ['member_id' => null, 'phone' => null, 'match_status' => 'member_not_found'];
    }

    $merged = false;
    $hops = 0;
    $db = getDB();
    $find = $db->prepare("SELECT id, phone, merged_into FROM members WHERE id = ? LIMIT 1");
    while (!empty($row['merged_into']) && $hops < PT_NOTIFY_MERGE_MAX_HOPS) {
        $find->execute([$row['merged_into']]);
        $next = $find->fetch(PDO::FETCH_ASSOC);
        if (!$next) break;
        $row = $next;
        $merged = true;
        $hops++;
    }
    if ($hops >= PT_NOTIFY_MERGE_MAX_HOPS && !empty($row['merged_into'])) {
        return ['member_id' => null, 'phone' => null, 'match_status' => 'member_not_found'];
    }

    $phone = trim((string)($row['phone'] ?? ''));
    if ($phone === '') {
        return ['member_id' => (int)$row['id'], 'phone' => null, 'match_status' => 'phone_empty'];
    }
    return [
        'member_id'    => (int)$row['id'],
        'phone'        => $phone,
        'match_status' => $merged ? 'merged_followed' : 'matched',
    ];
}

```javascript
(function () {
  'use strict';

  async function api(action, options = {}) {
    const url = '/api/notify.php?action=' + encodeURIComponent(action);
    const res = await fetch(url, {
      method: options.method || 'GET',
      headers: { 'Content-Type': 'application/json' },
      body: options.body ? JSON.stringify(options.body) : undefined,
    });
    const data = await res.json();
    if (!data.ok) throw new Error(data.message || 'API error');
    return data;
  }

  function fmtTime(s) { return s ? s : '—'; }

  function escapeHtml(s) {
    return String(s ?? '').replace(/[&<>"']/g, c => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[c]));
  }

  async function render(container) {
    container.innerHTML = '<div class="notify-page"><h2>알림톡 발송</h2><div id="notifyList">로딩 중…</div></div>';
    try {
      const { scenarios } = await api('list_scenarios');
      renderScenarios(scenarios);
    } catch (e) {
      document.getElementById('notifyList').innerHTML = `<div class="notify-error">${escapeHtml(e.message)}</div>`;
    }
  }

  function renderScenarios(scenarios) {
    const root = document.getElementById('notifyList');
    if (!scenarios.length) {
      root.innerHTML = '<div class="notify-empty">등록된 시나리오가 없습니다. <code>scenarios/</code> 디렉토리에 PHP 파일을 추가하세요.</div>';
      return;
    }
    root.innerHTML = scenarios.map(s => `
      <div class="notify-card" data-key="${escapeHtml(s.key)}">
        <div class="notify-card-head">
          <h3>${escapeHtml(s.name)} <span class="notify-key">${escapeHtml(s.key)}</span></h3>
          <label class="notify-toggle">
            <input type="checkbox" data-action="toggle" ${s.is_active ? 'checked' : ''}>
            <span>활성</span>
          </label>
        </div>
        <p class="notify-desc">${escapeHtml(s.description)}</p>
        <div class="notify-meta">
          <span>스케줄: <code>${escapeHtml(s.schedule)}</code></span>
          <span>cooldown: ${s.cooldown_hours}h</span>
          <span>마지막 실행: ${escapeHtml(fmtTime(s.last_run_at))} (${escapeHtml(s.last_run_status || '—')})</span>
          <span>다음 예정: ${escapeHtml(fmtTime(s.next_run_at))}</span>
        </div>
        <div class="notify-actions">
          <button class="btn btn-small" data-action="preview">미리보기</button>
          <button class="btn btn-small btn-outline" data-action="batches">배치이력</button>
        </div>
        <div class="notify-result"></div>
      </div>
    `).join('');

    root.querySelectorAll('.notify-card').forEach(card => {
      const key = card.dataset.key;
      card.querySelector('[data-action="toggle"]').addEventListener('change', e => onToggle(key, e.target.checked));
      card.querySelector('[data-action="preview"]').addEventListener('click', () => onPreview(card, key));
      card.querySelector('[data-action="batches"]').addEventListener('click', () => onBatches(card, key));
    });
  }

  async function onToggle(key, on) {
    try {
      await api('toggle', { method: 'POST', body: { key, is_active: on } });
    } catch (e) { alert('토글 실패: ' + e.message); }
  }

  async function onPreview(card, key) {
    const result = card.querySelector('.notify-result');
    result.innerHTML = '미리보기 생성 중…';
    try {
      const data = await api('preview', { method: 'POST', body: { key } });
      result.innerHTML = renderPreview(data);
      result.querySelector('[data-action="dispatch"]').addEventListener('click', () => onDispatch(card, data.preview_id));
    } catch (e) {
      result.innerHTML = `<div class="notify-error">${escapeHtml(e.message)}</div>`;
    }
  }

  function renderPreview(d) {
    const matched = d.matched || [];
    const unknown = d.unknown || [];
    const cooldown = d.cooldown_skip || [];
    const renderRow = (r, cls) => `<tr class="${cls}"><td>${escapeHtml(r.soritune_id || r.row_key)}</td><td>${escapeHtml(r.name)}</td><td>${escapeHtml(r.phone)}</td><td>${escapeHtml(r.match_status || '—')}</td></tr>`;
    return `
      <div class="notify-preview">
        <div class="notify-stats">
          <span>발송: <strong>${matched.length}</strong></span>
          <span class="notify-unknown">제외(unknown): <strong>${unknown.length}</strong></span>
          <span class="notify-cooldown">cooldown skip: <strong>${cooldown.length}</strong></span>
        </div>
        <table class="notify-table">
          <thead><tr><th>아이디</th><th>이름</th><th>phone</th><th>상태</th></tr></thead>
          <tbody>
            ${matched.map(r => renderRow(r, 'matched')).join('')}
            ${unknown.map(r => renderRow(r, 'unknown')).join('')}
            ${cooldown.map(r => renderRow(r, 'cooldown')).join('')}
          </tbody>
        </table>
        ${unknown.length > 0 ? '<div class="notify-warn">빨간색 행은 PT DB에 phone이 없어 발송되지 않습니다. 시트 또는 members 테이블을 수정하세요.</div>' : ''}
        <div class="notify-dispatch">
          <label><input type="checkbox" data-field="dry_run" checked> dry_run (테스트 발송, Solapi 호출 안 함)</label>
          <label>test_phone_override (선택, 입력 시 모든 메시지가 이 번호로 강제):
            <input type="text" data-field="test_phone" placeholder="01012345678">
          </label>
          <button class="btn btn-primary" data-action="dispatch" data-preview-id="${escapeHtml(d.preview_id)}">발송</button>
        </div>
      </div>
    `;
  }

  async function onDispatch(card, preview_id) {
    const result = card.querySelector('.notify-result');
    const dry_run = result.querySelector('[data-field="dry_run"]').checked ? 1 : 0;
    const test_phone = result.querySelector('[data-field="test_phone"]').value.trim();
    if (!confirm(`발송하시겠습니까? (dry_run=${dry_run}, test_phone=${test_phone || '없음'})`)) return;
    try {
      const body = { preview_id, dry_run };
      if (test_phone) body.test_phone_override = test_phone;
      const data = await api('send_now', { method: 'POST', body });
      result.innerHTML = `<div class="notify-success">배치 ${data.batch_id} 완료: 발송=${data.sent_count}, 실패=${data.failed_count}, 건너뜀=${data.skipped_count}</div>`;
    } catch (e) {
      result.innerHTML = `<div class="notify-error">${escapeHtml(e.message)}</div>`;
    }
  }

  async function onBatches(card, key) {
    const result = card.querySelector('.notify-result');
    result.innerHTML = '배치 이력 로딩 중…';
    try {
      const { batches } = await api('list_batches&key=' + encodeURIComponent(key));
      result.innerHTML = `
        <table class="notify-table">
          <thead><tr><th>ID</th><th>시작</th><th>종료</th><th>상태</th><th>대상/발송/실패</th><th>dry</th></tr></thead>
          <tbody>${batches.map(b => `
            <tr><td>${b.id}</td><td>${escapeHtml(b.started_at)}</td><td>${escapeHtml(b.finished_at || '—')}</td>
            <td>${escapeHtml(b.status)}</td><td>${b.target_count}/${b.sent_count}/${b.failed_count}</td>
            <td>${b.dry_run ? 'Y' : 'N'}</td></tr>`).join('')}</tbody>
        </table>`;
    } catch (e) {
      result.innerHTML = `<div class="notify-error">${escapeHtml(e.message)}</div>`;
    }
  }

  // PT SPA 등록. (실제 등록 방식은 Step 1에서 확인한 컨벤션을 따를 것)
  if (typeof App !== 'undefined' && App.pages) {
    App.pages.notify = { render };
  } else {
    window.NotifyPage = { render };
  }
})();
```
