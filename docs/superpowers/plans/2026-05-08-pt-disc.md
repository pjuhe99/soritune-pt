     * @param int[][] $answers 정확히 10개의 inner array, 각 inner = [D점수, I점수, S점수, C점수]
     * @throws InvalidArgumentException
     */
    public static function score(array $answers): array
    {
        $questions = self::questions();
        if (count($answers) !== count($questions)) {
            throw new InvalidArgumentException("answers length must be " . count($questions));
        }
        foreach ($answers as $i => $inner) {
            if (!is_array($inner) || count($inner) !== 4) {
                throw new InvalidArgumentException("answers[{$i}] must be an array of 4 ints");
            }
            $sorted = $inner;
            sort($sorted);
            if ($sorted !== [1, 2, 3, 4]) {
                throw new InvalidArgumentException("answers[{$i}] must be a permutation of {1,2,3,4}");
            }
        }

        $scores = ['D' => 0, 'I' => 0, 'S' => 0, 'C' => 0];
        $typeIdx = ['D' => 0, 'I' => 1, 'S' => 2, 'C' => 3];
        foreach ($answers as $inner) {
            foreach ($typeIdx as $t => $idx) {
                $scores[$t] += $inner[$idx];
            }
        }

        // ranks: 점수 내림차순, 동점 같은 rank
        $ranks = [];
        foreach ($scores as $t => $s) $ranks[] = ['type' => $t, 'score' => $s];
        usort($ranks, fn($a, $b) => $b['score'] <=> $a['score']);
        foreach ($ranks as $i => &$r) {
            if ($i === 0) {
                $r['rank'] = 1;
            } elseif ($r['score'] === $ranks[$i - 1]['score']) {
                $r['rank'] = $ranks[$i - 1]['rank'];
            } else {
                $r['rank'] = $i + 1;
            }
        }
        unset($r);

        // primary: 최고 점수, 동점 시 D > I > S > C
        $maxScore = max($scores);
        $primary = null;
        foreach (['D', 'I', 'S', 'C'] as $t) {
            if ($scores[$t] === $maxScore) {
                $primary = $t;
                break;
            }
        }

        $cat = self::categories()[$primary];

        return [
            'version'      => self::VERSION,
            'answers'      => $answers,
            'scores'       => $scores,
            'ranks'        => $ranks,
            'primary'      => $primary,
            'title'        => $cat['title'],
            'subtitle'     => $cat['subtitle'],
            'submitted_at' => date('c'),
        ];
    }
}
